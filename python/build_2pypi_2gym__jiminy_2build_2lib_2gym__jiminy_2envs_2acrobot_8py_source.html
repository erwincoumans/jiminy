17 class JiminyAcrobotGoalEnv(RobotJiminyGoalEnv):
18  """
19  @brief Implementation of a Gym environment for the Acrobot which is using
20  Jiminy Engine to perform physics computations and Gepetto-viewer for
21  rendering. It is a specialization of RobotJiminyGoalEnv. The acrobot
22  is a 2-link pendulum with only the second joint actuated. Initially,
23  both links point downwards. The goal is to swing the end-effector at
24  a height at least the length of one link above the base. Both links
25  can swing freely and can pass by each other, i.e. they don't collide
26  when they have the same angle.
27 
28  @details **STATE:**
29  The state consists of the sin() and cos() of the two rotational joint
30  angles and the joint angular velocities :
31  [cos(theta1) sin(theta1) cos(theta2) sin(theta2) thetaDot1 thetaDot2].
32  For the first link, an angle of 0 corresponds to the link pointing
33  downwards. The angle of the second link is relative to the angle of
34  the first link. An angle of 0 corresponds to having the same angle
35  between the two links. A state of [1, 0, 1, 0, ..., ...] means that
36  both links point downwards.
37 
38  **ACTIONS:**
39  The action is either applying +1, 0 or -1 torque on the joint between
40  the two pendulum links.
41 
42  @see R. Sutton: Generalization in Reinforcement Learning:
43  Successful Examples Using Sparse Coarse Coding (NIPS 1996)
44  @see R. Sutton and A. G. Barto:
45  Reinforcement learning: An introduction.
46  Cambridge: MIT press, 1998.
47  """

179  def _sample_goal(self):
180  """
181  @brief Samples a new goal and returns it.
182 
183  @details The goal is randomly sampled using a uniform
184  distribution between `0.2*self._tipPosZMax` and
185  `0.98*self._tipPosZMax`.
186 
187  @remark `self._tipPosZMax` can be overwritten to tweak the
188  difficulty of the problem. By default, it is the
189  maximum high of the highest joint that can be reached
190  by the pendulum.

193  """

197  def step(self, a):
198  """
199  @brief Run a simulation step for a given.
200 
201  @param[in] a The action to perform (in the action space rather than
202  the original torque space).
203 
204  @return The next observation, the reward, the status of the simulation
205  (done or not), and a dictionary of extra information
206  """
207  if self.continuous:
208  torque = a
209  else:
210  torque = self.AVAIL_TORQUE[a] * self.torque_mag
211 
212  # Add noise to the force action
213  if self.torque_noise_max > 0:
214  torque += self.np_random.uniform(-self.torque_noise_max, self.torque_noise_max)
215 
216  # Bypass 'self.engine_py.step' method and use direct assignment to max out the performances
217  self.engine_py._action[0] = torque
218  self.engine_py.step(dt_desired=self.dt)
219  self.state = self.engine_py.state
220 
221  # Get information
222  info, obs = self._get_info()
223  done = info['is_success']
224 
225  # Make sure the simulation is not already over
226  if done:
227  if self.steps_beyond_done is None:
228  self.steps_beyond_done = 0
229  else:
230  if self.steps_beyond_done == 0:
231  logger.warn("You are calling 'step()' even though this environment has already returned done = True. You should always call 'reset()' once you receive 'done = True' -- any further steps are undefined behavior.")
232  self.steps_beyond_done += 1
233 
234  # Compute the reward
235  reward = self.compute_reward(obs['achieved_goal'], self.goal, info)
236 
237  return obs, reward, done, info

242  @brief Get the observation associated to the current state of the model,

265  """
266  @brief Compute the reward at the current state.

315  def _get_obs(self):
316  """
317  @brief Get the current observation based on the current state of the model.
318 
319  @remark This is a hidden function that is not listed as part of the
320  member methods of the class. It is not intended to be called
321  manually.
322 
323  @return Dictionary with the current observation, achieved goal,
324  and desired goal.
325  """
326  theta1, theta2, theta1_dot, theta2_dot = self.state
327  theta1_dot = min(max(theta1_dot, -self.MAX_VEL_1), self.MAX_VEL_1)
328  theta2_dot = min(max(theta2_dot, -self.MAX_VEL_2), self.MAX_VEL_2)
329  observation = np.array([cos(theta1 + pi),
330  sin(theta1 + pi),
331  cos(theta2 + pi),
332  sin(theta2 + pi),
333  theta1_dot,
334  theta2_dot])
335 
336  achieved_goal = self._get_achieved_goal()
337  return {
338  'observation': observation,
339  'achieved_goal': achieved_goal.copy(),
340  'desired_goal': self.goal.copy(),
341  }
