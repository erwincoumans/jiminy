21 def update_quantities(jiminy_model,
22  position,
23  velocity=None,
24  acceleration=None,
25  update_physics=True,
26  update_com=False,
27  update_energy=False,
28  update_jacobian=False,
29  use_theoretical_model=True):
30  """
31  @brief Compute all quantities using position, velocity and acceleration
32  configurations.
33 
34  @details Run multiple algorithms to compute all quantities
35  which can be known with the model position, velocity
36  and acceleration configuration.
37 
38  This includes:
39  - body spatial transforms,
40  - body spatial velocities,
41  - body spatial drifts,
42  - body transform acceleration,
43  - body transform jacobians,
44  - center-of-mass position,
45  - center-of-mass velocity,
46  - center-of-mass drift,
47  - center-of-mass acceleration,
48  (- center-of-mass jacobian : No Python binding available so far),
49  - articular inertia matrix,
50  - non-linear effects (Coriolis + gravity)
51 
52  Computation results are stored in internal
53  data and can be retrieved with associated getters.
54 
55  @note This function modifies internal data.
56 
57  @param jiminy_model The Jiminy model
58  @param position Joint position vector
59  @param velocity Joint velocity vector
60  @param acceleration Joint acceleration vector
61 
62  @pre model_.nq == positionIn.size()
63  @pre model_.nv == velocityIn.size()
64  @pre model_.nv == accelerationIn.size()
65  """
66  if use_theoretical_model:
67  pnc_model = jiminy_model.pinocchio_model_th
68  pnc_data = jiminy_model.pinocchio_data_th
69  else:
70  pnc_model = jiminy_model.pinocchio_model
71  pnc_data = jiminy_model.pinocchio_data
72 
73  if (update_physics and update_com and \
74  update_energy and update_jacobian and \
75  velocity is not None):
76  pnc.computeAllTerms(pnc_model, pnc_data, position, velocity)
77  else:
78  if update_physics:
79  if velocity is not None:
80  pnc.nonLinearEffects(pnc_model, pnc_data, position, velocity)
81  pnc.crba(pnc_model, pnc_data, position)
82 
83  if update_jacobian:
84  # if update_com:
85  # pnc.getJacobianComFromCrba(pnc_model, pnc_data)
86  pnc.computeJointJacobians(pnc_model, pnc_data)
87 
88  if update_com:
89  if velocity is None:
90  pnc.centerOfMass(pnc_model, pnc_data, position)
91  elif acceleration is None:
92  pnc.centerOfMass(pnc_model, pnc_data, position, velocity)
93  else:
94  pnc.centerOfMass(pnc_model, pnc_data, position, velocity, acceleration)
95  else:
96  if velocity is None:
97  pnc.forwardKinematics(pnc_model, pnc_data, position)
98  elif acceleration is None:
99  pnc.forwardKinematics(pnc_model, pnc_data, position, velocity)
100  else:
101  pnc.forwardKinematics(pnc_model, pnc_data, position, velocity, acceleration)
102  pnc.framesForwardKinematics(pnc_model, pnc_data, position)
103 
104  if update_energy:
105  if velocity is not None:
106  pnc.kineticEnergy(pnc_model, pnc_data, position, velocity, False)
107  pnc.potentialEnergy(pnc_model, pnc_data, position, False)

109 def get_body_index_and_fixedness(jiminy_model, body_name, use_theoretical_model=True):
110  """
111  @brie Retrieve a body index in model and its fixedness from the body name.
112 
113  @details The method searchs bodyNames array and fixedBodyNames array
114  for the provided bodyName.
115 
116  @param jiminy_model The Jiminy model
117  @param body_name The name of the body for which index and fixedness are needed
118  @return body_id The index of the body whose name is bodyName in model
119  @return is_body_fixed Whether or not the body is considered as a fixed-body by the model
120  """
121  if use_theoretical_model:
122  pnc_model = jiminy_model.pinocchio_model_th
123  else:
124  pnc_model = jiminy_model.pinocchio_model
125 
126  frame_id = pnc_model.getFrameId(body_name)
127  parent_frame_id = pnc_model.frames[frame_id].previousFrame
128  parent_frame_type = pnc_model.frames[parent_frame_id].type
129  is_body_fixed = (parent_frame_type == FrameType.FIXED_JOINT)
130  if is_body_fixed:
131  body_id = frame_id
132  else:
133  body_id = pnc_model.frames[frame_id].parent
134 
135  return body_id, is_body_fixed

137 def get_body_world_transform(jiminy_model, body_name, use_theoretical_model=True):
138  """
139  @brief Get the transform from world frame to body frame for a given body.
140 
141  @details It is assumed that computeQuantities has been called.
142 
143  @param jiminy_model The Jiminy model
144  @param body_name The name of the body
145  @return transform Object representing the transform
146 
147  @pre body_name must be the name of a body existing in model.
148  """
149  if use_theoretical_model:
150  pnc_model = jiminy_model.pinocchio_model_th
151  pnc_data = jiminy_model.pinocchio_data_th
152  else:
153  pnc_model = jiminy_model.pinocchio_model
154  pnc_data = jiminy_model.pinocchio_data
155 
156  body_id, body_is_fixed = get_body_index_and_fixedness(jiminy_model, body_name)
157  if body_is_fixed:
158  transform_in_parent_frame = pnc_model.frames[body_id].placement
159  last_moving_parent_id = pnc_model.frames[body_id].parent
160  parent_transform_in_world = pnc_data.oMi[last_moving_parent_id]
161  transform = parent_transform_in_world * transform_in_parent_frame
162  else:
163  transform = pnc_data.oMi[body_id]
164 
165  return transform

237 def compute_freeflyer_state_from_fixed_body(jiminy_model, fixed_body_name, position,
238  velocity=None, acceleration=None,
239  use_theoretical_model=True):
240  """
241  @brief Fill rootjoint data from articular data when a body is fixed parallel to world.
242 
243  @details The hypothesis is that 'fixed_body_name' is fixed parallel to world frame.
244  So this method computes the position of freeflyer rootjoint in the fixed body frame.
245 
246  @note This function modifies internal data.
247 
248  @param model The Jiminy model
249  @param fixed_body_name The name of the body that is considered fixed parallel to world frame
250  @param[inout] position Must contain current articular data. The rootjoint data can
251  contain any value, it will be ignored and replaced
252  The method fills in rootjoint data
253  @param[inout] velocity Same as positionInOut but for velocity
254 
255  @pre Rootjoint must be a freeflyer
256  @pre positionInOut.size() == model_->nq
257  """
258  if use_theoretical_model:
259  pnc_model = jiminy_model.pinocchio_model_th
260  pnc_data = jiminy_model.pinocchio_data_th
261  else:
262  pnc_model = jiminy_model.pinocchio_model
263  pnc_data = jiminy_model.pinocchio_data
264 
265  position[:6].fill(0)
266  position[6] = 1.0
267  if velocity is not None:
268  velocity[:6].fill(0)
269  else:
270  velocity = np.zeros((pnc_model.nv,))
271  if acceleration is not None:
272  acceleration[:6].fill(0)
273  else:
274  acceleration = np.zeros((pnc_model.nv,))
275 
276  pnc.forwardKinematics(pnc_model, pnc_data, position, velocity, acceleration)
277  pnc.framesForwardKinematics(pnc_model, pnc_data, position)
278 
279  ff_M_fixed_body = get_body_world_transform(jiminy_model, fixed_body_name)
280  w_M_ff = ff_M_fixed_body.inverse()
281  base_link_translation = w_M_ff.translation
282  base_link_quaternion = Quaternion(w_M_ff.rotation)
283  position[:3] = base_link_translation
284  position[3:7] = base_link_quaternion.coeffs()
285 
286  ff_v_fixed_body = get_body_world_velocity(jiminy_model, fixed_body_name)
287  base_link_velocity = - ff_v_fixed_body
288  velocity[:6] = base_link_velocity.vector
289 
290  ff_a_fixedBody = get_body_world_acceleration(jiminy_model, fixed_body_name)
291  base_link_acceleration = - ff_a_fixedBody
292  acceleration[:6] = base_link_acceleration.vector

294 def retrieve_freeflyer(trajectory_data, roll_angle=0.0, pitch_angle=0.0):
295  """
296  @brief Retrieves the freeflyer positions and velocities. The reference frame is the support foot.
297  """
298  jiminy_model = trajectory_data['jiminy_model']
299  use_theoretical_model = trajectory_data['use_theoretical_model']
300  for s in trajectory_data['evolution_robot']:
301  # Extract the current position, velocity and acceleration
302  q, v, a = s.q.squeeze(), s.v.squeeze(), s.a.squeeze()
303 
304  # Compute freeflyer using support foot as reference frame.
305  compute_freeflyer_state_from_fixed_body(jiminy_model, s.support_foot,
306  q, v, a, use_theoretical_model)
307 
308  # Move freeflyer to take the foot angle into account.
309  # w: world frame, st: support foot frame, ff: freeflyer frame.
310  w_M_sf = XYZRPYToSe3(np.array([roll_angle, pitch_angle, 0.0, 0.0, 0.0, 0.0]))
311  sf_M_ff = XYZQUATToSe3(s.q[:7]) # Px, Py, Pz, Qx, Qy, Qz, Qw
312  w_M_ff = w_M_sf.act(sf_M_ff)
313  q[:3] = w_M_ff.translation
314  q[3:7] = Quaternion(w_M_ff.rotation).coeffs()

316 def compute_efforts(trajectory_data, index=(0, 0)):
317  """
318  @brief Compute the efforts in the trajectory using RNEA method.
319 
320  @param trajectory_data Sequence of States for which to compute the efforts.
321  @param index Index under which the efforts will be saved in the trajectory_data
322  (usually the couple of (roll, pitch) angles of the support foot).
323  """
324  jiminy_model = trajectory_data['jiminy_model']
325  use_theoretical_model = trajectory_data['use_theoretical_model']
326  if use_theoretical_model:
327  pnc_model = jiminy_model.pinocchio_model_th
328  pnc_data = jiminy_model.pinocchio_data_th
329  else:
330  pnc_model = jiminy_model.pinocchio_model
331  pnc_data = jiminy_model.pinocchio_data
332 
333 
334  root_joint_idx = pnc_model.getJointId('root_joint')
335  for s in trajectory_data['evolution_robot']:
336  # Apply a first run of rnea without explicit external forces
337  pnc.computeJointJacobians(pnc_model, pnc_data, s.q)
338  pnc.rnea(pnc_model, pnc_data, s.q, s.v, s.a)
339 
340  # Initialize vector of exterior forces to 0
341  fs = pnc.StdVec_Force()
342  fs.extend(len(pnc_model.names) * (pnc.Force.Zero(),))
343 
344  # Compute the force at the henkle level
345  support_foot_idx = pnc_model.frames[pnc_model.getBodyId(s.support_foot)].parent
346  fs[support_foot_idx] = pnc_data.oMi[support_foot_idx]\
347  .actInv(pnc_data.oMi[root_joint_idx]).act(pnc_data.f[root_joint_idx])
348 
349  # Recompute the efforts with RNEA and the correct external forces
350  s.tau[index] = pnc.rnea(pnc_model, pnc_data, s.q, s.v, s.a, fs)
351  s.f_ext[index] = fs[support_foot_idx].copy()
352 
353  # Add the force to the structure
354  s.f[index] = dict(list(zip(pnc_model.names, [f_ind.copy() for f_ind in pnc_data.f])))
355 
356  # Add the external force applied at the soles as if the floor was a parent of the soles
357  for joint in ['LeftSole', 'RightSole']:
358  ha_M_s = pnc_model.frames[pnc_model.getBodyId(joint)].placement
359  if s.support_foot == joint:
360  s.f[index][joint] = ha_M_s.actInv(s.f_ext[index])
361  else:
362  s.f[index][joint] = pnc.Force.Zero()
