99  def seed(self, seed=None):
100  """
101  @brief Specify the seed of the simulation.
102 
103  @details One must reset the simulation after updating the seed because
104  otherwise the behavior is undefined as it is not part of the
105  specification for Python Jiminy engines.
106 
107  @param[in] seed Desired seed as a Unsigned Integer 32bit
108  Optional: The seed will be randomly generated using np if omitted.
109 
110  @return Updated seed of the simulation
111  """
112  self.np_random, seed = seeding.np_random(seed)
113  self.engine_py.seed(seed)
114  self.state = self.engine_py.state
115  return [seed]

117  def reset(self):
118  """
119  @brief Reset the simulation.
120 
121  @details The initial state is randomly sampled using a uniform
122  distribution between `self.state_random_low` and
123  `self.state_random_high`.
124 
125  @remark `self.state_random_low` and `self.state_random_high` can be
126  overwritten. They are obtained by default by extracting the
127  information from the URDF file.
128 
129  @return Initial state of the simulation
130  """
131  self.state = self.np_random.uniform(low=self.state_random_low,
132  high=self.state_random_high)
133  self.engine_py.reset(self.state)
134  self.steps_beyond_done = None
135  return self._get_obs()
